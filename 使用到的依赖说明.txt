###	初始安装依赖

###	安装需要在node安装目录下的 node_modules\npm\npmrc文件添加代理
	registry = https://registry.npm.taobao.org
	disturl=https://npm.taobao.org/dist

###	安装express
	npm install express -i --save

### 后端语言国际化
	npm install i18n -i --save

### 前端语言国际化
	npm install vue-i18n -i --save

###	服务端socket.io即时通讯
	npm install socket.io -i --save
	
### 全局安装socket
    npm install vue-socket.io --save

### ip2region  通过ip查询地区
	npm install ip2region -i --save

### ua-parser-js 设备信息对象数据格式化
	npm install ua-parser-js -i --save

###	mysql数据库
	npm install mysql -i --save

###	node-cache内存缓存库
	npm install node-cache -i --save
	
### 异步控制流
    npm install async

###	防止xss攻击
	npm install xss -i --save

###	后端数据解密
	npm install node-rsa -i --save

###	生成token
	npm install jsonwebtoken -i --save

###	前端数据加密
	npm install jsencrypt -i --save

	npm install socket.io-client -i --save
